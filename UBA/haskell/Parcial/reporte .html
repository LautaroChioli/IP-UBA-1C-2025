cantidadNumerosAbundantes :: Integer -> Integer -> Integer
cantidadNumerosAbundantes d h
    | d == h && esAbundante d = 1
    | d == h = 0
    | esAbundante d = 1 + cantidadNumerosAbundantes (d + 1) h
    | otherwise = cantidadNumerosAbundantes (d + 1) h

divisoresPropios :: Integer -> [Integer]
divisoresPropios 0 = []
divisoresPropios n = divisoresPropiosAux n 1

divisoresPropiosAux :: Integer -> Integer -> [Integer]
divisoresPropiosAux n divisor
    | n == divisor = []
    | mod n divisor == 0 = divisor : divisoresPropiosAux n (divisor + 1)
    | otherwise = divisoresPropiosAux n (divisor + 1)

sumatoria :: [Integer] -> Integer
sumatoria [] = 0
sumatoria (x:xs) = x + sumatoria xs 

esAbundante :: Integer -> Bool
esAbundante n 
    | sumatoria (divisoresPropios n) > n = True
    | otherwise = False

cantidadParesColumna :: [[Integer]] -> Integer -> Integer
cantidadParesColumna matriz col = modulo (filtroPares (columnaDeMatriz matriz col))


columnaDeMatriz :: [[Integer]] -> Integer -> [Integer]
columnaDeMatriz [] _ = []
columnaDeMatriz (elegido:resto) col = columnaDeMatrizAux elegido 0 col : columnaDeMatriz resto col

columnaDeMatrizAux :: [Integer] -> Integer -> Integer -> Integer
columnaDeMatrizAux [] _ _ = 0
columnaDeMatrizAux (primer:resto) indice indiceBuscado
    | indiceReal == indiceBuscado = primer
    | otherwise = columnaDeMatrizAux resto indiceReal indiceBuscado
    where indiceReal = indice + 1

esPar :: Integer -> Bool
esPar n
    | mod n 2 == 0 = True
    | otherwise = False

filtroPares :: [Integer] -> [Integer]
filtroPares [] = []

    | esPar x = x : filtroPares xs
    | otherwise = filtroPares xs
